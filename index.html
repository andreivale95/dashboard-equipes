async function carregarDados() { const url = "https://script.google.com/macros/s/AKfycbzAqJ0uEAshdKoTRaw3N2rcyic13yHISCCM7m0OajH2SMwN76N33-RuOxTG5hj7yfdd/exec"; try { const response = await fetch(url); if (!response.ok) throw new Error("Erro ao carregar dados"); const dados = await response.json(); console.log(dados); // Inspecione os dados no console const dashboard = document.getElementById("dashboard"); dashboard.innerHTML = ""; dados.forEach(equipe => { // Verifique se o campo "status" existe e é uma string válida let status = equipe.status || "Status desconhecido"; if (typeof status !== "string") { status = "Status desconhecido"; } // Normaliza o status para usar no nome da classe CSS const statusClass = status.toLowerCase().replace(" ", "-"); const card = document.createElement("div"); card.className = `card card-${statusClass}`; // Cria o conteúdo do cartão dinamicamente let cardContent = `

${equipe.equipe}

`; // Itera sobre todas as chaves do objeto "equipe" for (const chave in equipe) { if (chave !== "equipe" && chave !== "status") { // Ignora "equipe" e "status" cardContent += `

${formatarChave(chave)}: ${equipe[chave]}

`; } } // Adiciona o status no final cardContent += `
Status: ${status}
`; card.innerHTML = cardContent; dashboard.appendChild(card); }); } catch (error) { const dashboard = document.getElementById("dashboard"); dashboard.innerHTML = `

Erro ao carregar dados

Mensagem: ${error.message}

Detalhes: Verifique o console para mais informações.

`; console.error("Erro detalhado:", error); } } // Função para formatar as chaves (ex: "auxiliar_1" → "Auxiliar 1") function formatarChave(chave) { return chave .replace(/_/g, " ") // Substitui underscores por espaços .replace(/\b\w/g, letra => letra.toUpperCase()); // Capitaliza a primeira letra de cada palavra } carregarDados(); setInterval(carregarDados, 30000); // Atualiza a cada 10 segundos
